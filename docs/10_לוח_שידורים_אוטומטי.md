## ⚙️ הגדרת GitHub Actions

### קובץ `.github/workflows/schedule-update.yml`

```yaml
name: Schedule Update

on:
  schedule:
    # רץ כל יום ראשון בשעה 00:00 UTC (02:00 שעון ישראל בחורף)
    - cron: '0 0 * * 0'
  workflow_dispatch:

jobs:
  update-schedule:
    runs-on: ubuntu-latest
    
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4
      
      - name: Setup Git
        run: |
          git config --global user.name "AnLoMinus"
          git config --global user.email "actions@github.com"
      
      - name: Update Schedule
        run: |
          echo "🔄 עדכון לוח שידורים אוטומטי"
          echo "📅 תאריך: $(date '+%Y-%m-%d')"
          # כאן ניתן להוסיף לוגיקה לעדכון לוח השידורים
          
      - name: Commit and Push
        run: |
          if [ -n "$(git status --porcelain)" ]; then
            git add .
            git commit -m "📅 Auto-update: עדכון לוח שידורים $(date '+%Y-%m-%d')"
            git push
          fi
```
